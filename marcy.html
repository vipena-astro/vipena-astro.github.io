#N>~1e7 <-- number of BHs
#steps~50000 <-- how many steps should I run my simulation for
# M=1e8 (Mo) <-- SMBH mass
# t <-- N x steps np.ndarray. t[j,i] <-- "the time of the simulation for the j-th BH at the i-th step"
# a <-- N x steps np.ndarray. a[j,i] <-- "the r-coord of the j-th BH at the i-th step"
# q <-- N x steps np.ndarray. q[j,i] <-- "the mass ratio m/M of the j-th BH at the i-th step"
#merged__BH_BH <-- list. Container for the IDs of the BHs that have merged up with some other BHs, and hence "exited" the simulation
#merged__IMBH_SMBH <-- list. Container for the IDs of the IMBHs that have inspiralled onto the SMBH, and hence "exited" too
#rk <-- (N,) np.ndarray. Container for the N Runge-Kutta *integrators* corresponding to each of the sBHs.
#a_antitrap <-- float. the radius of divergent migration in the AGN disc
#t_merger(a[j,i],q[j,i]) <-- function(float,float). the time (since the i-th step) over which the j-th IMBH will merge with the SMBH

############### MAINLOOP: ###############
for s in range(steps):
    ###1. MOVE THE BHs
    for j in range(N):
        ###1.0. SKIP THE BHs THAT HAVE EXITED THE SIMULATION
        if (j in merged__IMBH_SMBH) or (j in merged__BH_BH): #if it has already exited the simulation,
            continue #skip it
        
        ###1.1. STEP EACH RUNGE-KUTTA INTEGRATOR
        q[j,i]=q[j,i-1] #set j's mass at the i-th step to the same mass of j at the (i-1)-th step
        _=rk[j].step() #advance the Runge-Kutta integrator for the j-th BH one step
        t[j,i],a[j,i]=rk[j].t,rk[j].y #register the new time and position
        
        ###1.2. CHECK IF THE j-th BH INSPIRALS ONTO THE SMBH
        if q[j,i]<q_crit: #if it hasn't yet reached the critical mass for decoupling from the trap,
            condition=(a[j,i]<a_antitrap) and (t_merger(a[j,i],q[j,i])<AGN_dutycycle-t[j,i]) #yet it is somehow inner to the anti-trap, and if it's gonna merge within the AGN duty cycle;
        else: #or, if it decoupled from the trap already
            condition=t_merger(a[j,i],q[j,i])<AGN_dutycycle-t[j,i] #and it inspirals within an AGN duty cycle,
        if condition: #then
            merged__IMBH_SMBH.append(j) #consider it merged with the SMBH,
            continue #and don't analyse it further in this step.
        
    ###2. PERFORM THE sBH-sBH MERGERS
    ###2.1. CHECK *WHERE* THE MERGERS ARE, IF ANY
    A,B=np.meshgrid(a[:,i-1],a[:,i-1])
    distances=np.abs(A-B) #NxN np.ndarray <-- distances between the i-th and the j-th BH

    Hill_radii__vector=a[:,i-1]*(q[:,i-1]/3)**(1/3) #(N,) np.ndarray <-- the Hill radii of the i-th BH
    A,B=np.meshgrid(Hill_radii__vector,Hill_radii__vector)
    Hill_radii__matrix=np.where(A>B,A,B) #=max(r_H[i],r_H[j])

    lin=np.arange(N)
    A,B=np.meshgrid(lin,lin)
    triangle=A<B #NxN triangular Boolean mask
    mergers=triangle*(distances<Hill_radii__matrix) #"a merger between the i-th and the j-th BHs happens wherever d[i,j]<max(r_H[i],r_H[j])"

    ###2.2. GROUP THE BHs IN MERGERS
    merger_indices=np.where(mergers) #get the i,j pairs where a merger happens
    groups=[] #container for groups of merging BHs (allows for triple mergers)
    for sBH_a,sBH_b in zip(merger_indices[0],merger_indices[1]):
        added=False #so far, the pair (i,j)=(sBH_a,sBH_b) hasn't been added as a group. First we gotta check if we need to group it up with any prior pairs
        if (sBH_a in rk_done) or (sBH_a in rk_inspiralled) or (sBH_a in rk_failed): #if sBH_a had merged in a prior step, or
            continue
        elif (sBH_b in rk_done) or (sBH_b in rk_inspiralled) or (sBH_b in rk_failed):#if sBH_b had, then ignore this pair
            continue
        pair=sorted([sBH_a,sBH_b])
        for group in groups: #for each pre-recorded group A,
            if sBH_a in group or sBH_b in group: #if one of the BHs in the current pair B is already in A, then
                for sBH in pair:
                    if sBH not in group:
                        group.append(sBH) #add to A the BH from B that wasn't already in A
                added=True #flag the pair B as "added"
        if not added: #if B wasn't added to any pre-existing group, then
            groups.append(pair) #register B as a new group in "groups"

    ###2.3. PERFORM THE ACTUAL MERGING
    for group in groups: #for each group of merging BHs,
        j=group[0] #take the first BH as the group "leader". Programatically, j will accrete the other BHs in the group
        q[j,i]=sum([q[item,i-1] for item in group]) #set j's mass at this step to the sum of the masses in the group
        a[j,i]=sum([a[item,i-1]*q[item,i-1] for item in group])/sum([q[item,i-1] for item in group]) #set j's position at this step to the group's barycentre
        rk[j]=RK45(lambda t,a:f(t,a,q[j,i]),t[j,i],[a[j,i],],maxtime,first_step=first_step,max_step=max_step) #(re)set the Runge-Kutta integrator to a BH with j's parameters. This is necessary for calculating a[j,i+1]
        for item in group[1:]: #for every non-j BH in the group,
            merged__BH_BH.append(item) #flag them as "done", i.e. their RK integrators will be skipped in subsequent steps
        t_mergers.append(t[j,i]) #register the merger time
